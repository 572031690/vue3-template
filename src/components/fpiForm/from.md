## itemList 用法
# input
type: 'input'
label?: string // 标题
valueName: string // 变量名称
value?: any // 默认值
size?: 'large' | 'default' | 'small' // 单个input的size
clearable?: boolean // 是否可以一键清空
disabled?: boolean // 是否警用按钮
style?: any // 传入自定义样式
input?: 'password' // input内部的特殊type
placeholder?: string // 输入框占位文本
rules?: string | Object // 规则校验
